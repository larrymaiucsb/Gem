uint8_t temp_Read() {
	uint8_t value = 0;
	setInput(GPIOC, GPIO_PIN_0);
	for(int i = 0; i < 8; i++) {
		setOutput(GPIOC, GPIO_PIN_0);
		HAL_GPIO_WritePin(GPIOC, GPIO_PIN_0, 0);
		delay(2);
		setInput(GPIOC, GPIO_PIN_0);

			value |= 1 << i;
		}

uint8_t checkResponse() {
	uint8_t response = 0;
	delay(40);
	if(!(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0))) {
		delay(80);
		if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0)) {
			response = 1;
		} else {
			response = -1;
		}
	}

	return response;
}

uint8_t DHT22Read() {
	uint8_t i, j;
	for(j = 0; j < 8; j++) {
		while(!(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0)));
		delay(40);
		if(!(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0))) {
			i&= ~(1<<(7-j));
		} else {
			i |= (1<<(7-j));

		}
		while(!(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0)));

	}
	return i;
}
